PAYMENT SERVICE (Stripe + Chargily) // ───────────────────────────────────────────────────────────── // services/payment.service.ts import Stripe from 'stripe' const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!) export async function createPayment(orderId: string, method: string) { const order = await prisma.order.findUniqueOrThrow({ where: { id: orderId } }) if (method === 'STRIPE') { const intent = await stripe.paymentIntents.create({ amount: order.amountPaid, currency: 'usd', metadata: { orderId }, }) await prisma.payment.create({ data: { orderId, method: 'STRIPE', status: 'PENDING', amountRequested: order.amountPaid, currency: 'USD', stripePaymentIntentId: intent.id }, }) return { clientSecret: intent.client_secret } } // Chargily: CIB or Gold card (DZ) const chargilyMethod = method === 'CHARGILY_CIB' ? AI TRUST SERVICE (Claude Vision + Seller scoring) // ───────────────────────────────────────────────────────────── // services/trust.service.ts import Anthropic from '@anthropic-ai/sdk' const ai = new Anthropic() export async function scanListingImages(listingId: string): Promise { const images = await prisma.listingImage.findMany({ where: { listingId } }) if (!images.length) return 0 const scores = await Promise.all(images.map(async (img) => { const resp = await ai.messages.create({ model: 'claude-opus-4-5', max_tokens: 512, messages: [{ role: 'user', content: [ { type: 'image', source: { type: 'url', url: img.url } }, { type: 'text', text: `Analyze this phone listing image. Return ONLY valid JSON: {"fraudScore":0.0-1.0,"flags":[],"damageDetected":false,"isStockPhoto":false,"summary":""} Flags: stolen_photo | watermark | heavy_damage | mismatched_model | poor_quality` }, ], }], }) const json = JSON.parse((resp.content[0] as any).text) await prisma.listingImage.update({ where: { id: img.id }, data: { aiTagsJson: json } }) return json.fraudScore as number })) const avg = scores.reduce((a, b) => a + b, 0) / scores.length await prisma.listing.update({ where: { id: listingId }, data: { aiFraudScore: avg, aiVerified: avg < 0.3, status: avg > 0.7 ? 'UNDER_REVIEW' : 'ACTIVE' }, }) return avg } export async function recalcTrustScore(userId: string): Promise { const [reviews, orders, user] = await Promise.all([ prisma.review.findMany({ where: { revieweeId: userId }, select: { rating: true } }), prisma.order.findMany({ where: { sellerId: userId }, select: { status: true } }), prisma.user.findUniqueOrThrow({ where: { id: userId } }), ]) const avgRating = reviews.length ? reviews.reduce((s, r) => s + r.rating, 0) / reviews.length : 3 const total = orders.length || 1 const delivered = orders.filter(o => o.status === 'DELIVERED').length const disputed = orders.filter(o => o.status === 'DISPUTED').length const score = Math.min(100, Math.round( (avgRating / 5) * 40 + (delivered / total) * 30 + (user.isIdVerified ? 20 : 0) + Math.max(0, 10 - disputed * 5) )) await prisma.user.update({ where: { id: userId }, data: { trustScore: score } }) return score } // ───────────────────────────────────────────────────────────── // 11.
